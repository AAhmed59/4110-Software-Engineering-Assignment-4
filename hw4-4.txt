#Create an array of vowels
vowels = ["a", "e", "i", "o", "u", "y"]

#Ask user for input
letter = input("Please enter a single letter: ")

#Test to see if the letter is in the array or not
if (letter in vowels): #Returns true if the letter is in the array of vowels
    print("The letter", letter, "is a vowel.")
else: #If false is returned, then the letter is not in the array of vowels
    print("The letter", letter, "is not a vowel.")
